Python Implementation

# Install ShadowMask
pip install shadowmask

# Basic usage
from shadowmask import ShadowMask

# Initialize ShadowMask
sm = ShadowMask()

# Protect an image
protected_image = sm.protect_image(
    input_path="original.jpg",
    output_path="protected.jpg",
    methods=["alpha_attack", "adversarial", "encoder"],
    strength=0.8
)

# Batch processing
sm.protect_batch(
    input_dir="./images/",
    output_dir="./protected/",
    methods=["all"],
    preserve_metadata=True
)

# Advanced configuration
sm.configure({
    "alpha_intensity": 0.7,
    "adversarial_epsilon": 0.03,
    "encoder_layers": ["conv2d_1", "conv2d_3"],
    "diffusion_steps": 50
})

print(f"Protection applied: {protected_image.methods_used}")
print(f"Processing time: {protected_image.processing_time}s")

JavaScript/Node.js

// Install via npm
// npm install shadowmask-js

const ShadowMask = require('shadowmask-js');

// Initialize
const sm = new ShadowMask({
    apiKey: 'your-api-key',
    endpoint: 'https://api.shadowmask.dev'
});

// Protect image from file
async function protectImage() {
    try {
        const result = await sm.protectImage({
            inputPath: './original.jpg',
            outputPath: './protected.jpg',
            methods: ['alpha_attack', 'adversarial'],
            strength: 0.8
        });
        
        console.log('Protection applied:', result.methods);
        console.log('Processing time:', result.processingTime);
    } catch (error) {
        console.error('Protection failed:', error);
    }
}

// Protect image from canvas/blob
async function protectFromCanvas(canvas) {
    const blob = await new Promise(resolve => 
        canvas.toBlob(resolve, 'image/png')
    );
    
    const protected = await sm.protectBlob(blob, {
        methods: ['all'],
        format: 'png'
    });
    
    return protected;
}

protectImage();

cURL Examples

# Protect single image
curl -X POST https://api.shadowmask.dev/v1/protect \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: multipart/form-data" \
  -F "image=@original.jpg" \
  -F "methods=alpha_attack,adversarial" \
  -F "strength=0.8" \
  -o protected.jpg

# Get protection status
curl -X GET https://api.shadowmask.dev/v1/status/job_id \
  -H "Authorization: Bearer YOUR_API_KEY"

# Batch protection
curl -X POST https://api.shadowmask.dev/v1/batch \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "images": ["image1.jpg", "image2.png"],
    "methods": ["all"],
    "webhook_url": "https://your-app.com/webhook"
  }'

# Get API usage stats
curl -X GET https://api.shadowmask.dev/v1/usage \
  -H "Authorization: Bearer YOUR_API_KEY"
